## 快读
```cpp
template <typename T>
inline bool read(T &x) {
	x = 0; char c = getchar(); int f = 1;
	while (!isdigit(c) && (c != '-') && (c != EOF)) c = getchar();
	if (c == EOF) return 0;
	if (c == '-') f = -1, c = getchar();
	while (isdigit(c)) { x = x * 10 + (c & 15); c = getchar();}
	x *= f; return 1;
}

template <typename T, typename... Args>
inline bool read(T &x, Args &...args) {
	bool res = 1; 
	res &= read(x); 
	res &= read(args...); 
	return res;
}
```
